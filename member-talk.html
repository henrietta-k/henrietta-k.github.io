#A function for adding value labels to a bar chart

def make_chart(x, y):
  result = plt.bar(x, y)

  #Adding value labels 
  for i in range(len(x)):
    plt.text(i, y[i], y[i])

  return result

def add_labels(title, x, y):
  plt.title(title)
  plt.xlabel(x)
  plt.ylabel(y)
